Program 7: Remove Elements from an ArrayList
This program covers: remove(int index),remove(Object o),removeIf(Predicate),clear()

import java.util.ArrayList;

public class RemoveArrayListElements {
    public static void main(String[] args) {
        ArrayList<String> names = new ArrayList<>();

        names.add("Alice");
        names.add("Bob");
        names.add("Charlie");
        names.add("David");
        names.add("Bob");

        System.out.println("Original List: " + names);

        // Remove by index
        names.remove(2); // removes "Charlie"
        System.out.println("After removing index 2: " + names);

        // Remove by object
        names.remove("Bob"); // removes first "Bob"
        System.out.println("After removing 'Bob': " + names);

        // Remove using condition
        names.removeIf(name -> name.startsWith("D")); // removes "David"
        System.out.println("After removeIf condition: " + names);

        // Clear all
        names.clear();
        System.out.println("After clear(): " + names);
    }
}


Output:

Original List: [Alice, Bob, Charlie, David, Bob]
After removing index 2: [Alice, Bob, David, Bob]
After removing 'Bob': [Alice, David, Bob]
After removeIf condition: [Alice, Bob]
After clear(): []


Program 8: Sort and Reverse the ArrayList using Collections.sort() & Collections.reverse()

Program 9: Convert ArrayList to Array and vice versa

import java.util.ArrayList;
import java.util.Arrays;

public class ConvertArrayListArray {
    public static void main(String[] args) {
        // ArrayList to Array
        ArrayList<String> colors = new ArrayList<>();
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");

        String[] colorArray = new String[colors.size()];
        colors.toArray(colorArray);

        System.out.println("ArrayList: " + colors);
        System.out.println("Converted to Array: " + Arrays.toString(colorArray));

        // Array to ArrayList
        String[] animals = {"Dog", "Cat", "Elephant"};
        ArrayList<String> animalList = new ArrayList<>(Arrays.asList(animals));

        System.out.println("Array: " + Arrays.toString(animals));
        System.out.println("Converted to ArrayList: " + animalList);
    }
}

Output:

Converted to Array: [Red, Green, Blue]
Array: [Dog, Cat, Elephant]
Converted to ArrayList: [Dog, Cat, Elephant]

Program 11: Use retainAll, removeAll, and containsAll to compare and filter lists

  Keep only common elements with retainAll()
  Remove elements that exist in another list with removeAll()
  Check if one list contains all elements of another with containsAll()

import java.util.ArrayList;

public class ArrayListComparison {
    public static void main(String[] args) {
        ArrayList<String> listA = new ArrayList<>();
        listA.add("Apple");
        listA.add("Banana");
        listA.add("Cherry");

        ArrayList<String> listB = new ArrayList<>();
        listB.add("Banana");
        listB.add("Cherry");
        listB.add("Date");

        // retainAll(): keeps only the common elements
        ArrayList<String> common = new ArrayList<>(listA);
        common.retainAll(listB);
        System.out.println("Common Elements (retainAll): " + common);

        // removeAll(): removes all elements from listA that are in listB
        ArrayList<String> uniqueToA = new ArrayList<>(listA);
        uniqueToA.removeAll(listB);
        System.out.println("Elements unique to listA (removeAll): " + uniqueToA);

        // containsAll(): checks if listA contains all elements of listB
        boolean containsAll = listA.containsAll(listB);
        System.out.println("Does listA contain all elements of listB? " + containsAll);
    }
}

Output:

Common Elements (retainAll): [Banana, Cherry]
Elements unique to listA (removeAll): [Apple]
Does listA contain all elements of listB? false

Program 17–20: Custom Objects, Cloning, Sublist, and Iteration

import java.util.ArrayList;
import java.util.Collections;
import java.util.Iterator;
import java.util.List;
import java.util.ListIterator;

class Student {
    int id;
    String name;

    Student(int id, String name) {
        this.id = id;
        this.name = name;
    }

    public String toString() {
        return id + " - " + name;
    }
}

public class ArrayListOperations {
    public static void main(String[] args) {
        // 6. Clone ArrayList
        ArrayList<String> colors = new ArrayList<>();
        colors.add("Red");
        colors.add("Green");
        ArrayList<String> clonedColors = (ArrayList<String>) colors.clone();
        System.out.println("Cloned list: " + clonedColors);

        // 7. SubList
        colors.add("Blue");
        List<String> sub = colors.subList(0, 2); // index 0 and 1
        System.out.println("Sublist: " + sub);

        // 8. ArrayList of custom objects
        ArrayList<Student> students = new ArrayList<>();
        students.add(new Student(101, "Vimal"));
        students.add(new Student(102, "Raj"));
        System.out.println("Students: " + students);

        // 9. Using Collections.reverse
        Collections.reverse(colors);
        System.out.println("Reversed colors: " + colors);

        // 10. Iterate using all methods
        System.out.println("Iterating with for loop:");
        for (int i = 0; i < colors.size(); i++) {
            System.out.println(colors.get(i));
        }

        System.out.println("Iterating with for-each:");
        for (String color : colors) {
            System.out.println(color);
        }

        System.out.println("Iterating with Iterator:");
        Iterator<String> it = colors.iterator();
        while (it.hasNext()) {
            System.out.println(it.next());
        }

        System.out.println("Iterating with ListIterator (reverse):");
        ListIterator<String> listIt = colors.listIterator(colors.size());
        while (listIt.hasPrevious()) {
            System.out.println(listIt.previous());
        }

        // 10. Clear elements
        colors.clear();
        System.out.println("After clearing, colors: " + colors);
    }
}
