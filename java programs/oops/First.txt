class Car {
  string color;
  int speed;

void start() {
  System.out.println("Car is starting...");
}

void display() {
  System.out.println("Color " + color + "Speed " + speed + " km/hr");
}
}


public class myCar {
  public static void main(String[] args) {
     Car myCar = new Car();
     myCar.color = "red";
     myCar.speed = 120;
     myCar.start();
     myCar.display()
}
}

class Employee {
  String name;
  int id;

 Employee() {
   name = "unknown";
   id = 0;
 }

 Employee(String n , int i) {
   name = n;
   id = i;
 } 

 void show() {
   System.out.println("Name: " + name + " ID: " + id);
 }
}

public class ConstructorOverloadDemo {
  public static void main(String[] args) {
    Employee e1 = new Employee();
    Employee e2 = new Employee("Vimal" , 101);
    e1.show();
    e2.show();
}
}
