//Simple overloading

public class Calculator {
  void add() {
    System.out.println("Nothing to add");
  }

  void add(int a, int b){
     System.out.println("Sum(int) :" + (a+b));
   }

   void add(double a, double b) {
     System.out.println("Sum(double) :" + (a+b));
  }

  void add(int a, int b, int c){
    System.out.println("Sum(3 ints): " + (a+b+c));
  }

 public static void main(String[] args) {
    Calculator calc = new Calculator();
    calc.add();
    calc.add(5.5,4.5);
    calc.add(10,20);
    calc.add(1,2,3);
 }
}

//over riding

class Animal {
  void sound() {
    System.out.println("Animal makes sound");
  }
}


class Dog extends Animal {
  @override
  void sound() {
    System.out.println("Dog barks");
  }
}

public class OverrideDemo {
  public static void main(String[] args) {
    Animal a = new Dog();  // upcasting
    a.sound();    // dynamic method dispatch

 }
}

//Access Modifiers

class Animal {
  public void run(){
    System.out.println("Public: Running...");
  }

  protected void sleep() {
    System.out.println("Protected: Sleeping...");
  }

  void eat() {
    System.out.println("Default: Eating...");
  }

  private void breathe() {
    System.out.println("Private: Breathing...");
  }

 public void callPrivate() {
    breathe();
  }
}

class Dog extends Animal {
  void show() {
    run();   // ✅ public
    sleep();  // ✅ protected
    eat();    // ✅ default (same package)
    //breathe   ❌ Error – private not inherited
  }
}

public class AccessModifierDemo {
  public static void main(String[] args){
     Dog d = new Dog();
     d.run();
     d.sleep();
     d.eat();
     d.callPrivate();  // access private indirectly
  }
}
