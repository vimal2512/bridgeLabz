class Employee {
    String name;
    int salary;

    Employee(String name, int salary) {
        this.name = name;
        this.salary = salary;
    }

    void displayInfo() {
        System.out.println("Name: " + name);
        System.out.println("Salary: " + salary);
    }
}

class Manager extends Employee {
    int bonus;

    Manager(String name, int salary, int bonus) {
        super(name, salary);
        this.bonus = bonus;
    }

    void displayInfo() {
        super.displayInfo();
        System.out.println("Bonus: " + bonus);
        System.out.println("Total Pay: " + (salary + bonus));
    }
}

public class InheritanceUseCaseDemo {
    public static void main(String[] args) {
        Manager m = new Manager("Vimal Raj", 60000, 10000);
        m.displayInfo();
    }
}
