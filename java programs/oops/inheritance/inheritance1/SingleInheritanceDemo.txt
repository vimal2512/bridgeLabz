// Single Inheritance

class Animal {
  void eat() {
    System.out.println("Animal is eating...");
  }
}

class Dog Extends Animal {
   void bark() {
     System.out.println("Dog is barking...")
   }
}


public class SingleInheritanceDemo {
  public static void main(String[] args) {
      Dog dog = new Dog();
      dog.eat();
      dog.bark();
}
}
