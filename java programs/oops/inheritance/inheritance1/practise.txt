//multiple inheritance

interface Printable {
  void print();
}

interface Showable {
  void show();
}

class Document implements Printable,Showable {
    public void print() {
        System.out.println("Printing document");
    }

    public void show() {
        System.out.println("Showing document");
    }
}

public class MultipleInheritanceDemo {
  public static void main(String[] args) {
    Document doc = new Document();
    doc.print();
    doc.show();
  }
}

//Simple overloading

public class Calculator {
  void add() {
    System.out.println("Nothing to add");
  }

  void add(int a, int b) {
    System.out.println("Sum(double) :" + (a+b));
  }

  void add(double a, double b) {
    System.out.println("Sum(3 times) :" + (a+b));
  }

  void add(int a, int b, int c) {
     System.out.println("Sum(3 ints) :" + (a+b+c));
  }
}

public static void main(String[] args) {
   Calculator calc = new Calculator();
   calc.add();
   calc.add(5.5,5.5);
   calc.add(10,20);
   calc.ad(1,2,3);
}

//Overloading by changing order of parameters

public class OverloadOrder {
  void display(String name, int age) {
    System.out.println("Name: " + name + "Age: " + age);
  }

  void display(int age, String name) {
     System.out.println("Age: " + age + " Name: " + name);
  }

  public static void main(String[] args) {
     Overloader obj = new Overloader();
     obj.display("vimal" , 25);
     obj.display(28, "vimal");
  }
}
